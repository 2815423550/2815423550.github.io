*简易留言板

<body>
<h1>简易留言板</h1>
<div class="wrapper">
    <ul id="words"></ul>
</div>
<textarea id="msg"></textarea>

<script type="text/javascript">

    var ul = document.getElementById('words');
    var msg = document.getElementById('msg');
    var btn1 = document.getElementById('btn1');

    var liCount = 0; //用来记录留言数目的
    btn1.onclick = function () &#123;
        if(!msg.value)&#123;
            alert('留言板里没有内容')
        &#125;else&#123;
            /*将留言插入顶部的具体方法
            *1.判断ul中有无元素，没有则使用append，有则使用insertbefore
            * 2.插入信息同时插入一个span标签，用来设置关闭按钮
            * */
            var li = document.createElement('li');
            li.innerHTML = msg.value + '&nbsp;&nbsp;<span>X</span>';
            if(liCount == 0)
            &#123;
                ul.appendChild(li);
                liCount++;
            &#125;else&#123;
                ul.insertBefore(li,ul.childNodes[0]);
                liCount++;
            &#125;
            // 将输入框里的内容清空
            msg.value = '';
        &#125;
        oSpans = document.getElementsByTagName('span');
        for(var i = 0;i < oSpans.length;i++)&#123;
            oSpans[i].onclick = function()&#123;
            //this.parentNode 这里的this即是当前的span元素，span元素的父节点是li，意为移除该li元素
                ul.removeChild(this.parentNode);
                liCount--;
            &#125;
        &#125;
    &#125;;
</script>
